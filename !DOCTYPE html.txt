        <label for="tandatangan">Tanda Tangan:</label>
        <input type="text" id="tandatangan" name="tandatangan" required>

        <button type="submit">Kirim ke Google Sheet</button>
    </form>

    <!-- Elemen untuk menampilkan status pengiriman -->
    <div id="status"></div>

    <script>
        // Ambil elemen formulir
        const form = document.getElementById('acaraForm');
        const statusDiv = document.getElementById('status');
        
        // URL Web App dari Google Apps Script Anda
        const scriptURL = 'https://script.google.com/macros/s/AKfycby5Mh4hjk0qY2YYI9e3U9yWPqVtDLAECiSDofU20vSv3lcF8rPyi9CCdoG_og2Pd4zg/exec'; // <-- GANTI DENGAN URL ANDA!

        // Tambahkan event listener untuk saat formulir disubmit
        form.addEventListener('submit', e => {
            // Mencegah formulir mengirim data dengan cara default
            e.preventDefault();
            
            // Tampilkan status "Mengirim..."
            statusDiv.textContent = 'Mengirim data...';

            // Kirim data formulir menggunakan fetch() ke URL Apps Script
            fetch(scriptURL, { method: 'POST', body: new FormData(form)})
                .then(response => {
                    // Tampilkan pesan sukses dari Apps Script
                    statusDiv.textContent = 'Data berhasil dikirim!';
                    console.log('Success!', response);
                    // Kosongkan formulir setelah berhasil
                    form.reset(); 
                    // Hapus pesan sukses setelah 5 detik
                    setTimeout(() => { statusDiv.textContent = ''; }, 5000);
                })
                .catch(error => {
                    // Tampilkan pesan error jika gagal
                    statusDiv.textContent = 'Error! Data gagal dikirim.';
                    console.error('Error!', error.message);
                });
        });
    </script>
